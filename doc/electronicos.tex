\subsection{Divisor de voltaje}

{\justify
    \begin{equation*}
        I = \dfrac{V_{in}}{R_1 + R_2} \quad V_d = I R_2
    \end{equation*}

    \begin{equation*}
        V_d = \dfrac{V_{in}}{R_1 + R_2} R_2 \quad\rightarrow\quad V_d = \dfrac{V_{in} R_2}{R_1 + R_2}
    \end{equation*}

    \begin{equation*}
        V_d (R_1 + R_2) = V_{in} R_2 \quad\rightarrow\quad V_d R_1 + V_d R_2 = V_{in} R_2
    \end{equation*}

    \begin{equation*}
        V_d R_1 = V_{in} R_2 - V_d R_2 \quad\rightarrow\quad V_d R_1 = (V_{in} - V_d) R_2
    \end{equation*}

    \begin{equation*}
        R_1 = \dfrac{(V_{in} - V_d) R_2}{V_d} \quad\quad \dfrac{V_d R_1}{V_{in} - V_d} = R_2
    \end{equation*}
}
